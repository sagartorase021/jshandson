        const findSum = (n) => {
            var sum = 0;

Find the sum of the digits of a given number.

const Number_Sum = (N) => 
        {
            var n_sum = 0;
            var rem = 0;
            while(N>0)
            {
              rem = N%10;
              n_sum = n_sum + rem;
              N = Math.trunc(N/10);
            }
            return n_sum;
        };
